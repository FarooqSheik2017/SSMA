	PROCEDURE SP_LOAD_HEALTHCARE_ENCOUNTER
	AS
	BEGIN 

		INSERT INTO SAIKIRAN.FACT_ENCOUNTER_TGT(
			ENCOUNTER_ID, 
			PATIENT_ID, 
			PROVIDER_ID, 
			DIAGNOSIS_ID, 
			ADMIT_DATE, 
			DISCHARGE_DATE, 
			ENCOUNTER_TYPE, 
			LENGTH_OF_STAY, 
			TOTAL_CHARGES, 
			INSURANCE_TYPE, 
			DISCHARGE_DISPOSITION, 
			READMISSION_FLAG, 
			HCPCS_CODE)
			SELECT 
				s.ENCOUNTER_ID, 
				s.PATIENT_ID, 
				s.PROVIDER_ID, 
				s.DIAGNOSIS_ID, 
				s.ADMIT_DATE, 
				s.DISCHARGE_DATE, 
				UPPER(s.ENCOUNTER_TYPE), 
				
					CASE 
						WHEN s.ADMIT_DATE IS NOT NULL AND s.DISCHARGE_DATE IS NOT NULL THEN CAST(s.DISCHARGE_DATE AS DATE) - CAST(s.ADMIT_DATE AS DATE)
						ELSE NULL
					END AS LENGTH_OF_STAY, 
				s.TOTAL_CHARGES, 
				s.INSURANCE_TYPE, 
				s.DISCHARGE_DISPOSITION, 
				
					CASE 
						WHEN UPPER(s.READMISSION_FLAG) IN ( 'Y', 'YES', '1', 'TRUE' ) THEN 'Y'
						ELSE 'N'
					END AS READMISSION_FLAG, 
				s.HCPCS_CODE
			FROM SAIKIRAN.STG_ENCOUNTER  s 
				JOIN SAIKIRAN.DIM_PATIENT  p  ON p.PATIENT_ID = s.PATIENT_ID 
				JOIN SAIKIRAN.DIM_PROVIDER  pr  ON pr.PROVIDER_ID = s.PROVIDER_ID AND UPPER(pr.IS_ACTIVE) IN ( 'Y', 'YES', '1' ) 
				JOIN SAIKIRAN.DIM_DIAGNOSIS  d  ON d.DIAGNOSIS_ID = s.DIAGNOSIS_ID
			WHERE NOT EXISTS 
				(
					SELECT 1
					FROM SAIKIRAN.FACT_ENCOUNTER_TGT  f
					WHERE f.ENCOUNTER_ID = s.ENCOUNTER_ID
				);

		COMMIT;

	END;

CREATE PROCEDURE SAIKIRAN.SP_LOAD_HEALTHCARE_ENCOUNTER  
AS 
	BEGIN

		INSERT SAIKIRAN.FACT_ENCOUNTER_TGT(
		   ENCOUNTER_ID, 
		   PATIENT_ID, 
		   PROVIDER_ID, 
		   DIAGNOSIS_ID, 
		   ADMIT_DATE, 
		   DISCHARGE_DATE, 
		   ENCOUNTER_TYPE, 
		   LENGTH_OF_STAY, 
		   TOTAL_CHARGES, 
		   INSURANCE_TYPE, 
		   DISCHARGE_DISPOSITION, 
		   READMISSION_FLAG, 
		   HCPCS_CODE)
		   SELECT 
		   	S.ENCOUNTER_ID, 
		   	S.PATIENT_ID, 
		   	S.PROVIDER_ID, 
		   	S.DIAGNOSIS_ID, 
		   	S.ADMIT_DATE, 
		   	S.DISCHARGE_DATE, 
		   	upper(S.ENCOUNTER_TYPE), 
		   	CASE 
		   		WHEN S.ADMIT_DATE IS NOT NULL AND S.DISCHARGE_DATE IS NOT NULL THEN ssma_oracle.datediff(CAST(S.DISCHARGE_DATE AS datetime2(0)), CAST(S.ADMIT_DATE AS datetime2(0)))
		   		ELSE NULL
		   	END AS LENGTH_OF_STAY, 
		   	S.TOTAL_CHARGES, 
		   	S.INSURANCE_TYPE, 
		   	S.DISCHARGE_DISPOSITION, 
		   	CASE 
		   		WHEN upper(S.READMISSION_FLAG) IN ( 'Y', 'YES', '1', 'TRUE' ) THEN 'Y'
		   		ELSE 'N'
		   	END AS READMISSION_FLAG, 
		   	S.HCPCS_CODE
		   FROM SAIKIRAN.STG_ENCOUNTER  AS S 
		   	JOIN SAIKIRAN.DIM_PATIENT  AS P 
		   	ON P.PATIENT_ID = S.PATIENT_ID 
		   	JOIN SAIKIRAN.DIM_PROVIDER  AS PR 
		   	ON PR.PROVIDER_ID = S.PROVIDER_ID AND upper(PR.IS_ACTIVE) IN ( 'Y', 'YES', '1' ) 
		   	JOIN SAIKIRAN.DIM_DIAGNOSIS  AS D 
		   	ON D.DIAGNOSIS_ID = S.DIAGNOSIS_ID
		   WHERE NOT EXISTS 
		   	(
		   		SELECT 1 AS expr
		   		FROM SAIKIRAN.FACT_ENCOUNTER_TGT  AS F
		   		WHERE F.ENCOUNTER_ID = S.ENCOUNTER_ID
		   	)

		IF @@TRANCOUNT > 0
			COMMIT TRANSACTION 

	END
