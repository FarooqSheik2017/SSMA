CREATE OR REPLACE TRIGGER SAIKIRAN.TRG_VAL_PROVIDER_ID
	BEFORE INSERT OR UPDATE OF  PROVIDER_ID
	ON "SAIKIRAN"."DIM_PROVIDER"
	FOR EACH ROW
	DECLARE
		v_count NUMBER;
	BEGIN 

		/* 1. Validate Not Null*/
		IF :NEW.PROVIDER_ID IS NULL THEN 
			RAISE_APPLICATION_ERROR(-20001, 'PROVIDER_ID cannot be null.');
		END IF;

		/* 2. Validate Uniqueness*/
		SELECT COUNT(*)
		INTO v_count
		FROM "SAIKIRAN"."DIM_PROVIDER"
		WHERE PROVIDER_ID = :NEW.PROVIDER_ID AND ROWID != NVL(:OLD.ROWID, '0');

		/* Ensure we aren't checking the record against itself during updates*/
		IF v_count > 0 THEN 
			RAISE_APPLICATION_ERROR(-20002, 'PROVIDER_ID must be unique. Value ' || :NEW.PROVIDER_ID || ' already exists.');
		END IF;

	END;

CREATE TRIGGER SAIKIRAN.InsteadOfInsertOn$DIM_PROVIDER
	ON SAIKIRAN.DIM_PROVIDER
	 INSTEAD OF INSERT
	AS 
		BEGIN

			SET  NOCOUNT  ON

			/* column variables declaration*/
			DECLARE
				@NEW$0 uniqueidentifier, 
				@NEW$PROVIDER_ID varchar(10), 
				@NEW$NPI varchar(15), 
				@NEW$PROVIDER_NAME varchar(100), 
				@NEW$SPECIALTY varchar(50),

				@NEW$LICENSE_STATE varchar(2), 
				@NEW$LICENSE_EXPIRY datetime2(0), 
				@NEW$PHONE_NUMBER varchar(20), 
				@NEW$EMAIL varchar(100), 
				@NEW$IS_ACTIVE varchar(5)

			DECLARE
				 ForEachInsertedRowTriggerCursor CURSOR LOCAL FORWARD_ONLY READ_ONLY FOR 
					SELECT 
						ROWID, 
						PROVIDER_ID, 
						NPI, 
						PROVIDER_NAME, 
						SPECIALTY, 
						FACILITY_NAME, 
						LICENSE_STATE, 
						LICENSE_EXPIRY, 
						PHONE_NUMBER, 
						EMAIL, 
						IS_ACTIVE
					FROM inserted

			OPEN ForEachInsertedRowTriggerCursor

			FETCH ForEachInsertedRowTriggerCursor
				 INTO 
					@NEW$0, 
					@NEW$PROVIDER_ID, 
					@NEW$NPI, 
					@NEW$PROVIDER_NAME, 
					@NEW$SPECIALTY, 
					@NEW$FACILITY_NAME, 
					@NEW$LICENSE_STATE, 
					@NEW$LICENSE_EXPIRY, 
					@NEW$PHONE_NUMBER, 
					@NEW$EMAIL, 
					@NEW$IS_ACTIVE

			WHILE @@fetch_status = 0
			
				BEGIN

					/* row-level triggers implementation: begin*/
					BEGIN
						IF (UPDATE(PROVIDER_ID))
							BEGIN

								DECLARE
									/*
									*   SSMA warning messages:
									*   O2SS0356: Conversion from NUMBER datatype can cause data loss.
									*/

									@V_COUNT float(53)

								/* 1. Validate Not Null*/
								IF @NEW$PROVIDER_ID IS NULL
									BEGIN

										DECLARE
											@db_raise_application_error_message nvarchar(4000)

										SET @db_raise_application_error_message = N'ORA' + CAST(-20001 AS nvarchar) + N': ' + N'PROVIDER_ID cannot be null.'
; 
										THROW 59998, @db_raise_application_error_message, 1

									END

								/* 
								*   SSMA error messages:
								*   O2SS0083: Identifier ":OLD".ROWID cannot be converted because it was not resolved.

								/* 2. Validate Uniqueness*/
								SELECT @V_COUNT = count_big(*)
								FROM SAIKIRAN.DIM_PROVIDER
								WHERE DIM_PROVIDER.PROVIDER_ID = @NEW$PROVIDER_ID AND ROWID != isnull([:OLD].ROWID, '0')
								*/



								/* Ensure we aren't checking the record against itself during updates*/
								IF @V_COUNT > 0
									BEGIN

										DECLARE
											@db_raise_application_error_message$2 nvarchar(4000)

										SET @db_raise_application_error_message$2 = N'ORA' + CAST(-20002 AS nvarchar) + N': ' + ('PROVIDER_ID must be unique. Value ' + ISNULL(@NEW$PROVIDER_ID, '') + ' already exists.')
; 
										THROW 59998, @db_raise_application_error_message$2, 1

									END

							END
					END
					/* row-level triggers implementation: end*/

					/* DML-operation emulation*/
					INSERT SAIKIRAN.DIM_PROVIDER(
					   ROWID, 
					   PROVIDER_ID, 
					   NPI, 
					   PROVIDER_NAME,

					   FACILITY_NAME, 
					   LICENSE_STATE, 
					   LICENSE_EXPIRY, 
					   PHONE_NUMBER, 
					   EMAIL, 
					   IS_ACTIVE)
					   VALUES (
					   	@NEW$0, 
					   	@NEW$PROVIDER_ID, 
					   	@NEW$NPI, 
					   	@NEW$PROVIDER_NAME, 
					   	@NEW$SPECIALTY, 
					   	@NEW$FACILITY_NAME, 
					   	@NEW$LICENSE_STATE, 
					   	@NEW$LICENSE_EXPIRY, 
					   	@NEW$PHONE_NUMBER, 
					   	@NEW$EMAIL, 
					   	@NEW$IS_ACTIVE)

					FETCH ForEachInsertedRowTriggerCursor
						 INTO 
							@NEW$0, 
							@NEW$PROVIDER_ID, 
							@NEW$NPI, 
							@NEW$PROVIDER_NAME, 
							@NEW$SPECIALTY, 
							@NEW$FACILITY_NAME, 
							@NEW$LICENSE_STATE, 
							@NEW$LICENSE_EXPIRY, 
							@NEW$PHONE_NUMBER, 
							@NEW$EMAIL, 
							@NEW$IS_ACTIVE

				END

			CLOSE ForEachInsertedRowTriggerCursor

			DEALLOCATE ForEachInsertedRowTriggerCursor

		END
