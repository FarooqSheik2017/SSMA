CREATE VIEW SAIKIRAN.DIM_DIAGNOSIS_MV
WITH SCHEMABINDING 
AS 
	SELECT DIM_DIAGNOSIS.DIAGNOSIS_ID, count_big(DISTINCT DIM_DIAGNOSIS.DIAGNOSIS_CATEGORY) AS COUNT_DIAGNOSIS, COUNT_BIG(*) AS COUNT_BIG_COLUMN_FOR_GROUP_BY
	FROM SAIKIRAN.DIM_DIAGNOSIS
	GROUP BY DIM_DIAGNOSIS.DIAGNOSIS_ID
GO
