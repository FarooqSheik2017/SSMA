CREATE OR REPLACE PROCEDURE          DISPLAY_OUT_PATIENTS AS 
-- 1. Declare the Cursor-- Joins Patient and Encounter to get meaningful display data    
CURSOR cur_outpatients IS SELECT 
            p.PATIENT_ID,
            p.FIRST_NAME || ' ' || p.LAST_NAME AS FULL_NAME,
            e.ENCOUNTER_ID,
            e.ADMIT_DATE,
            e.ENCOUNTER_TYPE
        FROM 
            SAIKIRAN.DIM_PATIENT p
        JOIN 
            SAIKIRAN.FACT_ENCOUNTER e ON p.PATIENT_ID = e.PATIENT_ID
        WHERE 
            UPPER(e.ENCOUNTER_TYPE) = 'OUTPATIENT';
    -- 2. Declare a record variable to hold cursor rows    
	rec_outpatient cur_outpatients%ROWTYPE;
	BEGIN    DBMS_OUTPUT.PUT_LINE('--- Outpatient List ---');
    DBMS_OUTPUT.PUT_LINE('ID | Name | Encounter ID | Date');
    DBMS_OUTPUT.PUT_LINE('-----------------------------------');
    -- 3. Open the Cursor    
	OPEN cur_outpatients;
    LOOP-- 4. Fetch data into the record variable
	FETCH cur_outpatients INTO rec_outpatient;
        
        -- Exit when no more rows are found        
		EXIT WHEN cur_outpatients%NOTFOUND;
        -- 5. Display the data        
		DBMS_OUTPUT.PUT_LINE(
            rec_outpatient.PATIENT_ID || ' | ' || 
            rec_outpatient.FULL_NAME || ' | ' || 
            rec_outpatient.ENCOUNTER_ID || ' | ' || 
            TO_CHAR(rec_outpatient.ADMIT_DATE, 'DD-MON-YYYY')
        );
    END LOOP;
    -- 6. Close the Cursor    
	CLOSE cur_outpatients;
EXCEPTION WHEN OTHERS THEN IF cur_outpatients%ISOPEN THEN            CLOSE cur_outpatients;
        END IF;
        DBMS_OUTPUT.PUT_LINE('Error occurred: ' || SQLERRM);END;
