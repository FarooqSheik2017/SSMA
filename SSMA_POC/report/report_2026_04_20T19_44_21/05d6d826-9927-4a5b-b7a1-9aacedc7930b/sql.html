CREATE OR REPLACE 
	PROCEDURE DISPLAY_OUT_PATIENTS
	AS
	/* 1. Declare the Cursor-- Joins Patient and Encounter to get meaningful display data*/
	
		CURSOR cur_outpatients
			IS 
				SELECT 
					p.PATIENT_ID, 
					p.FIRST_NAME || ' ' || p.LAST_NAME AS FULL_NAME, 
					e.ENCOUNTER_ID, 
					e.ADMIT_DATE, 
					e.ENCOUNTER_TYPE
				FROM SAIKIRAN.DIM_PATIENT  p 
					JOIN SAIKIRAN.FACT_ENCOUNTER  e  ON p.PATIENT_ID = e.PATIENT_ID
				WHERE UPPER(e.ENCOUNTER_TYPE) = 'OUTPATIENT';
		/* 2. Declare a record variable to hold cursor rows*/
		rec_outpatient cur_outpatients%ROWTYPE;
	BEGIN 

		DBMS_OUTPUT.PUT_LINE('--- Outpatient List ---');

		DBMS_OUTPUT.PUT_LINE('ID | Name | Encounter ID | Date');

		DBMS_OUTPUT.PUT_LINE('-----------------------------------');

		/* 3. Open the Cursor*/
		OPEN cur_outpatients;

		LOOP 
			/* 4. Fetch data into the record variable*/
			FETCH cur_outpatients
				INTO rec_outpatient;
			/* Exit when no more rows are found*/
			EXIT WHEN cur_outpatients%NOTFOUND;
			/* 5. Display the data*/
			DBMS_OUTPUT.PUT_LINE(
				rec_outpatient.PATIENT_ID
				 || 
				' | '
				 || 
				rec_outpatient.FULL_NAME
				 || 
				' | '
				 || 
				rec_outpatient.ENCOUNTER_ID
				 || 
				' | '
				 || 
				TO_CHAR(rec_outpatient.ADMIT_DATE, 'DD-MON-YYYY'));
		END LOOP;

		/* 6. Close the Cursor*/
		CLOSE cur_outpatients;

		EXCEPTION 
			WHEN OTHERS THEN 

				IF cur_outpatients%ISOPEN THEN 
					CLOSE cur_outpatients;
				END IF;

				DBMS_OUTPUT.PUT_LINE('Error occurred: ' || SQLERRM);


	END;

CREATE PROCEDURE SAIKIRAN.DISPLAY_OUT_PATIENTS  
AS 
	BEGIN

		BEGIN TRY

			/* 2. Declare a record variable to hold cursor rows*/
			DECLARE
				@REC_OUTPATIENT$PATIENT_ID varchar(max), 
				@REC_OUTPATIENT$FULL_NAME varchar(max), 
				@REC_OUTPATIENT$ENCOUNTER_ID varchar(max), 
				@REC_OUTPATIENT$ADMIT_DATE datetime2(0), 
				@REC_OUTPATIENT$ENCOUNTER_TYPE varchar(max)

			PRINT '--- Outpatient List ---'

			PRINT 'ID | Name | Encounter ID | Date'

			PRINT '-----------------------------------'

			DECLARE
				 CUR_OUTPATIENTS CURSOR LOCAL FOR 
					SELECT 
						P.PATIENT_ID, 
						ISNULL(P.FIRST_NAME, '') + ' ' + ISNULL(P.LAST_NAME, '') AS FULL_NAME, 
						E.ENCOUNTER_ID, 
						E.ADMIT_DATE, 
						E.ENCOUNTER_TYPE
					FROM SAIKIRAN.DIM_PATIENT  AS P 
						JOIN SAIKIRAN.FACT_ENCOUNTER  AS E 
						ON P.PATIENT_ID = E.PATIENT_ID
					WHERE upper(E.ENCOUNTER_TYPE) = 'OUTPATIENT'

			/* 3. Open the Cursor*/
			OPEN CUR_OUTPATIENTS

			WHILE 1 = 1
			
				BEGIN

					/* 4. Fetch data into the record variable*/
					FETCH CUR_OUTPATIENTS
						 INTO 
							@REC_OUTPATIENT$PATIENT_ID, 
							@REC_OUTPATIENT$FULL_NAME, 
							@REC_OUTPATIENT$ENCOUNTER_ID, 
							@REC_OUTPATIENT$ADMIT_DATE, 
							@REC_OUTPATIENT$ENCOUNTER_TYPE

					/*
					*   SSMA warning messages:
					*   O2SS0113: The value of @@FETCH_STATUS might be changed by previous FETCH operations on other cursors, if the cursors are used simultaneously.
					*/

					/* Exit when no more rows are found*/
					IF @@FETCH_STATUS <> 0
						BREAK

					/* 5. Display the data*/
					PRINT 
						ISNULL(@REC_OUTPATIENT$PATIENT_ID, '')
						 + 
						' | '
						 + 
						ISNULL(@REC_OUTPATIENT$FULL_NAME, '')
						 + 
						' | '
						 + 
						ISNULL(@REC_OUTPATIENT$ENCOUNTER_ID, '')
						 + 
						' | '
						 + 
						ISNULL(ssma_oracle.to_char_date(@REC_OUTPATIENT$ADMIT_DATE, 'DD-MON-YYYY'), '')

				END

			/* 6. Close the Cursor*/
			CLOSE CUR_OUTPATIENTS

			DEALLOCATE CUR_OUTPATIENTS

		END TRY

		BEGIN CATCH

			DECLARE
				@errornumber int

			SET @errornumber = ERROR_NUMBER()

			DECLARE
				@errormessage nvarchar(4000)

			SET @errormessage = ERROR_MESSAGE()

			DECLARE
				@exceptionidentifier nvarchar(4000)

			SELECT @exceptionidentifier = ssma_oracle.db_error_get_oracle_exception_id(@errormessage, @errornumber)

			BEGIN

				IF CURSOR_STATUS('local', N'CUR_OUTPATIENTS') > -1
					BEGIN

						CLOSE CUR_OUTPATIENTS

						DEALLOCATE CUR_OUTPATIENTS

					END

				PRINT 'Error occurred: ' + ISNULL(ssma_oracle.db_error_sqlerrm_0(@exceptionidentifier, @errornumber), '')

			END

		END CATCH

	END
