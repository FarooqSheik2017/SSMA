CREATE OR REPLACE VIEW VW_ADMITTED_PATIENTS(PATIENT_ID, FIRST_NAME, LAST_NAME, GENDER, PHONE_NUMBER, ENCOUNTER_ID, ADMIT_DATE, ENCOUNTER_TYPE) AS
SELECT DISTINCT    
    P.PATIENT_ID,
    P.FIRST_NAME,
    P.LAST_NAME,
    P.GENDER,
    P.PHONE_NUMBER,
    E.ENCOUNTER_ID,
    E.ADMIT_DATE,
    E.ENCOUNTER_TYPE
    --E.FACILITY_NAME
    FROM 
    SAIKIRAN.DIM_PATIENT P INNER JOIN 
    SAIKIRAN.FACT_ENCOUNTER E ON P.PATIENT_ID = E.PATIENT_ID WHERE 
    E.ADMIT_DATE IS NOT NULL
;
