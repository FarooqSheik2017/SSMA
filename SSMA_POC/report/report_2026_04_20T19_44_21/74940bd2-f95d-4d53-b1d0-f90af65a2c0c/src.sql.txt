CREATE OR REPLACE VIEW VW_PATIENT_DIAGNOSIS(PATIENT_ID, FIRST_NAME, LAST_NAME, DIAGNOSIS_ID, ICD_CODE, DIAGNOSIS_DESC, DIAGNOSIS_CATEGORY, SEVERITY, ENCOUNTER_ID, ADMIT_DATE) AS
SELECT 
    P.PATIENT_ID,
    P.FIRST_NAME,
    P.LAST_NAME,
    D.DIAGNOSIS_ID,
    D.ICD_CODE,
    D.DIAGNOSIS_DESC,
    D.DIAGNOSIS_CATEGORY,
    D.SEVERITY,
    E.ENCOUNTER_ID,
    E.ADMIT_DATE FROM 
    SAIKIRAN.DIM_PATIENT P JOIN 
    SAIKIRAN.FACT_ENCOUNTER E ON P.PATIENT_ID = E.PATIENT_ID JOIN 
    SAIKIRAN.DIM_DIAGNOSIS D ON E.DIAGNOSIS_ID = D.DIAGNOSIS_ID
;
